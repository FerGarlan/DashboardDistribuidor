    <script>

        Chart.register(ChartDataLabels);

        Chart.defaults.color = '#94a3b8';

        Chart.defaults.borderColor = '#1e293b';

 

        const dropzone = document.getElementById('dropzone');

        const csvInput = document.getElementById('csvInput');

        const uploadText = document.getElementById('uploadText');

        const periodFilter = document.getElementById('periodFilter');

       

        const palette = ['#3b82f6', '#2563eb', '#1d4ed8', '#1e40af', '#1e3a8a', '#172554', '#0c4a6e', '#075985'];

        let charts = {};

        let rawData = [];

 

        dropzone.onclick = () => csvInput.click();

        csvInput.onchange = (e) => handleFile(e.target.files[0]);

        periodFilter.onchange = () => processData(rawData);

 

        function handleFile(file) {

            if (!file) return;

            uploadText.innerText = "Procesando...";

            Papa.parse(file, {

                header: true,

                skipEmptyLines: true,

                transformHeader: h => h.trim(),

                complete: function(results) {

                    rawData = results.data;

                    periodFilter.disabled = false;

                    processData(rawData);

                    uploadText.innerHTML = `<span class="text-blue-400">✓ ${rawData.length.toLocaleString()} registros</span>`;

                }

            });

        }

 

        function cleanNumber(val) {

            if (!val) return 0;

            let str = val.toString().replace(/[$\s]/g, '');

            if (str.includes(',') && str.includes('.')) {

                if (str.indexOf(',') < str.indexOf('.')) str = str.replace(/,/g, '');

                else str = str.replace(/\./g, '').replace(',', '.');

            } else if (str.includes(',')) str = str.replace(',', '.');

            return parseFloat(str) || 0;

        }

 

        function parseDate(fechaStr) {

            if (!fechaStr) return null;

            const sep = fechaStr.includes('-') ? '-' : '/';

            const parts = fechaStr.split(sep);

            if (parts.length < 3) return null;

            return parts[0].length === 4

                ? new Date(parts[0], parts[1]-1, parts[2])

                : new Date(parts[2], parts[1]-1, parts[0]);

        }

 

        function formatUSD(val) {

            return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }).format(val);

        }

 

        function processData(data) {

            const months = periodFilter.value;

            let filtered = data;

 

            let maxDate = null;

            data.forEach(d => {

                const dt = parseDate(d['Fecha Emision']);

                if (dt && (!maxDate || dt > maxDate)) maxDate = dt;

            });

 

            if (months !== 'all' && maxDate) {

                const cutoff = new Date(maxDate);

                cutoff.setMonth(cutoff.getMonth() - parseInt(months));

                filtered = data.filter(d => {

                    const dt = parseDate(d['Fecha Emision']);

                    return dt && dt >= cutoff;

                });

            }

 

            let neta = 0, pvp = 0, tpa = 0, count = 0;

            const resP = {}, resG = {}, resPr = {}, resM = new Array(12).fill(0);

 

            filtered.forEach(d => {

                const vPvp = cleanNumber(d['Venta PVP Voucher']);

                const vNeta = cleanNumber(d['Venta Neta Voucher']);

                const vTpa = cleanNumber(d['Fee TPA Voucher']);

                const pais = d['Pais POS'] || 'N/D';

                const gsa = d['GSA'] || 'Directo';

                const prod = d['Plan Comercial'] || 'Otros';

 

                count++;

                neta += vNeta;

                pvp += vPvp;

                tpa += vTpa;

 

                const dt = parseDate(d['Fecha Emision']);

                if (dt) resM[dt.getMonth()] += vNeta;

 

                resP[pais] = (resP[pais] || 0) + vNeta;

                resG[gsa] = (resG[gsa] || 0) + vNeta;

                resPr[prod] = (resPr[prod] || 0) + vNeta;

            });

 

            const contrib = neta - tpa;

            const cPos = pvp - neta;

 

            document.getElementById('kpi-pvp').innerText = formatUSD(pvp);

            document.getElementById('kpi-neta').innerText = formatUSD(neta);

            document.getElementById('kpi-perc-neta').innerText = pvp > 0 ? ((neta/pvp)*100).toFixed(1) + '% s/ PVP' : '0%';

            document.getElementById('kpi-contrib').innerText = formatUSD(contrib);

            document.getElementById('kpi-perc').innerText = neta > 0 ? ((contrib/neta)*100).toFixed(1) + '% de Margen' : '0%';

            document.getElementById('kpi-costo-pos').innerText = formatUSD(cPos);

            document.getElementById('kpi-perc-pos').innerText = pvp > 0 ? ((cPos/pvp)*100).toFixed(1) + '% s/ PVP' : '0%';

            document.getElementById('kpi-tpa').innerText = formatUSD(tpa);

            document.getElementById('kpi-perc-tpa').innerText = pvp > 0 ? ((tpa/pvp)*100).toFixed(1) + '% s/ PVP' : '0%';

            document.getElementById('kpi-ticket-prom').innerText = formatUSD(count > 0 ? neta/count : 0);

            document.getElementById('kpi-contrib-unit').innerText = formatUSD(count > 0 ? contrib/count : 0);

            document.getElementById('kpi-vouchers').innerText = count.toLocaleString();

 

            updateChart('productChart', resPr);

            updateChart('countryChart', resP);

            updateChart('gsaChart', resG);

            updateTrend(resM);

        }

 

        function updateChart(id, dataMap) {

            if (charts[id]) charts[id].destroy();

            const entries = Object.entries(dataMap).sort((a,b) => b[1]-a[1]).slice(0,5);

            const total = entries.reduce((a,b) => a + b[1], 0);

 

            charts[id] = new Chart(document.getElementById(id), {

                type: 'doughnut',

                data: {

                    labels: entries.map(e => `${e[0]}: ${total > 0 ? ((e[1]/total)*100).toFixed(1) : 0}%`),

                    datasets: [{ data: entries.map(e => e[1]), backgroundColor: palette, borderWidth: 2, borderColor: '#0f172a' }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    cutout: '75%',

                    plugins: { legend: { position: 'bottom', labels: { boxWidth: 8, font: { size: 10 } } }, datalabels: { display: false } }

                }

            });

        }

 

        function updateTrend(data) {

            if (charts.trend) charts.trend.destroy();

            charts.trend = new Chart(document.getElementById('trendChart'), {

                type: 'bar',

                data: {

                    labels: ["Ene", "Feb", "Mar", "Abr", "May", "Jun", "Jul", "Ago", "Sep", "Oct", "Nov", "Dic"],

                    datasets: [{ label: 'Venta Neta', data: data, backgroundColor: '#3b82f6', borderRadius: 6 }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: { legend: { display: false }, datalabels: { display: false } },

                    scales: {

                        y: { beginAtZero: true, grid: { color: '#1e293b' }, ticks: { callback: v => '$' + (v >= 1000 ? v/1000 + 'k' : v) } },

                        x: { grid: { display: false } }

                    }

                }

            });

        }
